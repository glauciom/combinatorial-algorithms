 523:        matrix_add(n_2, n_2, B, bx+n_2, by, bs, B, bx+n_2, by+n_2, bs, b_cum, 0, 0, n_2);
 524:        strassen_mult(n_2, a_cum, 0, 0, n_2, b_cum, 0, 0, n_2, p7, 0, 0, n_2, d+1, qd, s);
 525:    }
 526:     }
 527: 
 528:     /* c11 = p1 + p4 - p5 + p7 */
 529:     matrix_add(n_2, n_2, p1, 0, 0, n_2, p4, 0, 0, n_2, C, cx, cy, cs);
 530:     matrix_sub(n_2, n_2, C, cx, cy, cs, p5, 0, 0, n_2, C, cx, cy, cs);
 531:     matrix_add(n_2, n_2, C, cx, cy, cs, p7, 0, 0, n_2, C, cx, cy, cs);
 532: 
 533:     /* c12 = p3 + p5 */
 534:     matrix_add(n_2, n_2, p3, 0, 0, n_2, p5, 0, 0, n_2, C, cx, cy+n_2, cs);
 535: 
 536:     /* c21 = p2 + p4 */
 537:     matrix_add(n_2, n_2, p2, 0, 0, n_2, p4, 0, 0, n_2, C, cx+n_2, cy, cs);
 538: 
 539:     /* c22 = p1 + p3 - p2 + p6 */
 540:     matrix_add(n_2, n_2, p1, 0, 0, n_2, p3, 0, 0, n_2, C, cx+n_2, cy+n_2, cs);
 541:     matrix_sub(n_2, n_2, C, cx+n_2, cy+n_2, cs, p2, 0, 0, n_2, C, cx+n_2, cy+n_2, cs);
 542:     matrix_add(n_2, n_2, C, cx+n_2, cy+n_2, cs, p6, 0, 0, n_2, C, cx+n_2, cy+n_2, cs);
 543: }
 544: 
 545: /******************************************************************************************\
 546:  * Perform A x B => C  for square submatrices of A, B, and C assuming the submatrix 
 547:  * dimension is divisible by two.
 548:  *
 549:  *      n_2 = n/2 = order of partitioned matrices
 550:  *
 551:  *          ------------- -------------   -------------
 552:  *      n/2 ! a11 ! a12 ! ! b11 ! b12 !   ! c11 ! c12 ! n/2
 553:  *          !-----!-----!*!-----!-----! = !-----!-----!
 554:  *      n/2 ! a21 ! a22 ! ! b21 ! b22 !   ! c21 ! c22 ! n/2
 555:  *          ------------- -------------   -------------
 556:  *            n/2   n/2     n/2   n/2       n/2   n/2
 557:  *
 558:  *      algorithm:
 559:  *
 560:  *      p1 = (a11+a22)*(b11+b22)
 561:  *      p2 = (a21+a22)*b11
 562:  *      p3 = a11*(b12-b22)
 563:  *      p4 = a22*(b21-b11)
 564:  *      p5 = (a11+a12)*b22
 565:  *      p6 = (a21-a11)*(b11+b12)
 566:  *      p7 = (a12-a22)*(b21+b22)
 567:  *
 568:  *      c11 = p1+p4-p5+p7
 569:  *      c12 = p3+p5
 570:  *      c21 = p2+p4
 571:  *      c22 = p1+p3-p2+p6
 572:  *
 573:  *      where each matrix multiplication is implemented as
 574:  *      a recursive call to strassen_mult.
 575:  *
 576:  * 
 577: \******************************************************************************************/
 578: 
 579: void 
 580: strassen_mult(
 581:        int n,                      /* dimensions of A, B, and C submatrices */
 582:        double *A, int ax, int ay, int as,   /* (ax,ay) = origin of A submatrix for multiplicand */
 583:        double *B, int bx, int by, int bs,   /* (bx,by) = origin of B submatrix for multiplicand */
 584:        double *C, int cx, int cy, int cs,   /* (cx,cy) = origin of C submatrix for result */
 585:        int d,                      /* current depth of strassen's recursion */
 586:        int qd,                      /* target depth for taskq recursion */
 587:        int s                     /* Strassen's recursion limit for array dimensions */
 588: )
 589: {
 590:     if (n < s) {
 591: 
 592:    matrix_mult(n, n, n, A, ax, ay, as, B, bx, by, bs, C, cx, cy, cs, d);
 593: 
 594:     }
 595:     else {
 596: 
 597: #if DEBUG 
 598:    #pragma omp critical(debug)
 599:    {
 600:         int i;
 601:         for (i = 1; i <= d; i++) printf(" ");
 602:         printf("strassen_mult: (t%d) (%dx%d)*(%dx%d); depth %d; A[%d..%d][%d..%d](%dx%d) *"
 603:                " B[%d..%d][%d..%d](%dx%d) => C[%d..%d][%d..%d](%dx%d)\n", 
 604:           omp_get_thread_num(), n, n, n, n, d, 
 605:           ax, ax+n-1, ay, ay+n-1, as, as,  
 606:           bx, bx+n-1, by, by+n-1, bs, bs,
 607:           cx, cx+n-1, cy, cy+n-1, cs, cs
 608:         );
 609:    }
 610: #endif
 611:    if ( d <= qd && parallel ) {
 612: 
 613:             strassen_mult_par(n, A, ax, ay, as, B, bx, by, bs, C, cx, cy, cs, d, qd, s);
 614: 
 615:    } else {
 616:        int n_2 = n >> 1;
 617:        double *work;
 618:        double *a_cum, *b_cum;
 619:        double *p1, *p2, *p3, *p4, *p5, *p6, *p7;
 620: 
 621:        work  = (double *) alloca (sizeof(double) * n_2 * n_2 * 9);
 622:        a_cum = work; 
 623:        b_cum = a_cum + n_2 * n_2;
 624:        p1    = b_cum + n_2 * n_2;
 625:        p2    = p1 + n_2 * n_2;
 626:        p3    = p2 + n_2 * n_2;
 627:        p4    = p3 + n_2 * n_2;
 628:        p5    = p4 + n_2 * n_2;
 629:        p6    = p5 + n_2 * n_2;
 630:        p7    = p6 + n_2 * n_2;
 631: 
 632:          /* p1 = (a11 + a22) x (b11 + b22) */
 633:        matrix_add(n_2, n_2, A, ax, ay, as, A, ax+n_2, ay+n_2, as, a_cum, 0, 0, n_2);
 634:        matrix_add(n_2, n_2, B, bx, by, bs, B, bx+n_2, by+n_2, bs, b_cum, 0, 0, n_2);
 635:        strassen_mult(n_2, a_cum, 0, 0, n_2, b_cum, 0, 0, n_2, p1, 0, 0, n_2, d+1, qd, s);
 636: 
 637:        /* p2 = (a21 + a22) x b11 */
 638:        matrix_add(n_2, n_2, A, ax+n_2, ay, as, A, ax+n_2, ay+n_2, as, a_cum, 0, 0, n_2);
 639:        strassen_mult(n_2, a_cum, 0, 0, n_2, B, bx, by, bs, p2, 0, 0, n_2, d+1, qd, s);
 640:        
 641:        /* p3 = a11 x (b12 - b22) */
 642:        matrix_sub(n_2, n_2, B, bx, by+n_2, bs, B, bx+n_2, by+n_2, bs, b_cum, 0, 0, n_2);
 643:        strassen_mult(n_2, A, ax, ay, as, b_cum, 0, 0, n_2, p3, 0, 0, n_2, d+1, qd, s);
 644: 
 645:        /* p4 = a22 x (b21 - b11) */
 646:        matrix_sub(n_2, n_2, B, bx+n_2, by, bs, B, bx, by, bs, b_cum, 0, 0, n_2);
 647:        strassen_mult(n_2, A, ax+n_2, ay+n_2, as, b_cum, 0, 0, n_2, p4, 0, 0, n_2, d+1, qd, s);
 648: 
 649:        /* p5 = (a11 + a12) x b22 */
 650:        matrix_add(n_2, n_2, A, ax, ay, as, A, ax, ay+n_2, as, a_cum, 0, 0, n_2);
 651:        strassen_mult(n_2, a_cum, 0, 0, n_2, B, bx+n_2, by+n_2, bs, p5, 0, 0, n_2, d+1, qd, s);
 652:        
 653:        /* p6 = (a21 - a11) x (b11 + b12) */
 654:        matrix_sub(n_2, n_2, A, ax+n_2, ay, as, A, ax, ay, as, a_cum, 0, 0, n_2);
 655:        matrix_add(n_2, n_2, B, bx, by, bs, B, bx, by+n_2, bs, b_cum, 0, 0, n_2);
 656:        strassen_mult(n_2, a_cum, 0, 0, n_2, b_cum, 0, 0, n_2, p6, 0, 0, n_2, d+1, qd, s);
 657: 
 658:        /* p7 = (a12 - a22) x (b21 + b22) */
 659:        matrix_sub(n_2, n_2, A, ax, ay+n_2, as, A, ax+n_2, ay+n_2, as, a_cum, 0, 0, n_2);
 660:        matrix_add(n_2, n_2, B, bx+n_2, by, bs, B, bx+n_2, by+n_2, bs, b_cum, 0, 0, n_2);
 661:        strassen_mult(n_2, a_cum, 0, 0, n_2, b_cum, 0, 0, n_2, p7, 0, 0, n_2, d+1, qd, s);
 662: 
 663:        /* c11 = p1 + p4 - p5 + p7 */
 664:        matrix_add(n_2, n_2, p1, 0, 0, n_2, p4, 0, 0, n_2, C, cx, cy, cs);
 665:        matrix_sub(n_2, n_2, C, cx, cy, cs, p5, 0, 0, n_2, C, cx, cy, cs);
 666:        matrix_add(n_2, n_2, C, cx, cy, cs, p7, 0, 0, n_2, C, cx, cy, cs);
 667: 
 668:        /* c12 = p3 + p5 */
 669:        matrix_add(n_2, n_2, p3, 0, 0, n_2, p5, 0, 0, n_2, C, cx, cy+n_2, cs);
 670: 
 671:        /* c21 = p2 + p4 */
 672:        matrix_add(n_2, n_2, p2, 0, 0, n_2, p4, 0, 0, n_2, C, cx+n_2, cy, cs);
 673: 
 674:        /* c22 = p1 + p3 - p2 + p6 */
 675:        matrix_add(n_2, n_2, p1, 0, 0, n_2, p3, 0, 0, n_2, C, cx+n_2, cy+n_2, cs);
 676:        matrix_sub(n_2, n_2, C, cx+n_2, cy+n_2, cs, p2, 0, 0, n_2, C, cx+n_2, cy+n_2, cs);
 677:        matrix_add(n_2, n_2, C, cx+n_2, cy+n_2, cs, p6, 0, 0, n_2, C, cx+n_2, cy+n_2, cs);
 678:    }
 679:     }
 680:     return;
 681: }

 684: /******************************************************************************************\
 685:  * Perform A x B => C  for square A, B, and C matrices.
 686:  *
 687:  * First, decompose as follows:
 688:  * 
 689:  *             ss    bs      ss    bs        ss    bs
 690:  *          ------------- -------------   -------------
 691:  *       ss ! a11 ! a12 ! ! b11 ! b12 !   ! c11 ! c12 ! ss
 692:  *          !-----!-----!*!-----!-----! = !-----!-----!
 693:  *       bs ! a21 ! a22 ! ! b21 ! b22 !   ! c21 ! c22 ! bs
 694:  *          ------------- -------------   -------------
 695:  *             ss    bs      ss    bs        ss    bs
 696:  *
 697:  *
 698:  * Where ss = the maximum value <= n such that ss is divisible by 2 ^ fd
 699:  *       bs = the remaining border dimension for a22, b22, and c22
 700:  * 
 701:  *       we are assuming that bs << ss to maximize parallelism
 702:  *
 703:  * Then:
 704:  *       c11 = strassen(a11 x b11) + (a12 x b21) 
 705:  *
 706:  *       c12 = (a11,a12) x (b12)
 707:  *                         (b22)
 708:  *
 709:  *       c21 = (a21,a22) x (b11)
 710:  *                         (b21)
 711:  *
 712:  *       c22 = (a21,a22) x (b12)
 713:  *                         (b22)
 714:  *
 715: \******************************************************************************************/
 716: 
 717: void 
 718: strassen_main(
 719:        int n,            /* dimensions of A, B, and C submatrices */
 720:        double *A,   /* A submatrix for multiplicand */
 721:        double *B,   /* B submatrix for multiplicand */
 722:        double *C,   /* C submatrix for result */
 723:        int qd,      /* target depth for taskq recursion */
 724:        int s,      /* Strassen's recursion limit for array dimensions */
 725:        int fd       /* depth of recursion for strassen's algorithm */
 726: )
 727: {
 728:     int ss;    /* strassen's piece size */
 729:     int bs;    /* remaining border size */    
 730: 
 731:     ss = (n>>fd)<<fd;
 732:     bs = n - ss;
 733: 
 734: #if DEBUG
 735:      printf("strassen_main: (t%d) (%dx%d)*(%dx%d); depth %d; A[%d..%d][%d..%d](%dx%d) *"
 736:        " B[%d..%d][%d..%d](%dx%d) => C[%d..%d][%d..%d](%dx%d)\n", 
 737:        omp_get_thread_num(), n, n, n, n, 0, 
 738:        0, size-1, 0, size-1, size, size,  
 739:        0, size-1, 0, size-1, size, size,
 740:        0, size-1, 0, size-1, size, size
 741:      );
 742:      printf("               strassen size = (%dx%d); strassen ops ~= %d;"
 743:             " matmul ops ~= %d\n", 
 744:             ss, ss, ss*ss*ss, ss*bs*ss + 2*ss*n*bs + bs*n*bs);
 745: #endif
 746: 
 747:     #pragma omp parallel taskq if(parallel) default(none) \
 748: 	    shared(A, B, C, bs, qd, s, ss, size)
 749:     {
 750:    #pragma omp task
 751:    {
 752:        if (bs == 0) {
 753: #if DEBUG_PAR 
 754:       #pragma omp critical(debug)
 755:       {
 756:            printf("task: (t%d) (%dx%d)*(%dx%d); depth %d; C[%d..%d][%d..%d]\n",
 757:              omp_get_thread_num(), ss, ss, ss, ss, 0, 0, ss-1, 0, ss-1 
 758:            );
 759:       }
 760: #endif
 761:       /* compute C <= A x B (in parallel) */
 762:       strassen_mult(ss, A, 0, 0, size, B, 0, 0, size, C, 0, 0, size, 1, qd, s);
 763: 
 764:        } else {
 765:       double *tmp; 
 766:       tmp = (double *) alloca (ss * ss * sizeof(double));
 767: 
 768:            #pragma omp taskq
 769:       {
 770:           /* compute c11 <= a11 x b11 (in parallel) */
 771:           #pragma omp task
 772:           {
 773: #if DEBUG_PAR 
 774:          #pragma omp critical(debug)
 775:          {
 776:               printf("task: (t%d) (%dx%d)*(%dx%d); depth %d; c11[%d..%d][%d..%d]\n",
 777:                 omp_get_thread_num(), ss, ss, ss, ss, 0, 0, ss-1, 0, ss-1 
 778:               );
 779:          }
 780: #endif
 781:          strassen_mult(ss, A, 0, 0, size, B, 0, 0, size, C, 0, 0, size, 1, qd, s);
 782:           }
 783: 
 784:           /* compute tmp <= a12 x b21 */
 785:           #pragma omp task
 786:           {
 787: #if DEBUG_PAR 
 788:          #pragma omp critical(debug)
 789:          {
 790:               printf("task: (t%d) (%dx%d)*(%dx%d); depth %d; tmp[%d..%d][%d..%d]\n",
 791:                 omp_get_thread_num(), ss, bs, bs, ss, 0, 0, ss-1, 0, ss-1 
 792:               );
 793:          }
 794: #endif
 795:          matrix_mult(ss, bs, ss, A, 0, ss, size, B, ss, 0, size, tmp, 0, 0, ss, 1); 
 796:           }
 797: 
 798:           /* compute c12 <= (a11,a12) x (b12,b22)T */
 799:           #pragma omp task
 800:           {
 801: #if DEBUG_PAR 
 802:          #pragma omp critical(debug)
 803:          {
 804:               printf("task: (t%d) (%dx%d)*(%dx%d); depth %d; c12[%d..%d][%d..%d]\n",
 805:                 omp_get_thread_num(), ss, n, n, bs, 0, 0, ss-1, ss, ss+bs-1 
 806:               );
 807:          }
 808: #endif
 809:          matrix_mult(ss, n, bs, A, 0, 0, size, B, 0, ss, size, C, 0, ss, size, 1);
 810:           }
 811: 
 812:           /* compute c21 <= (a21,a22) x (b11,b21)T */
 813:           #pragma omp task
 814:           {
 815: #if DEBUG_PAR 
 816:          #pragma omp critical(debug)
 817:          {
 818:               printf("task: (t%d) (%dx%d)*(%dx%d); depth %d; c21[%d..%d][%d..%d]\n",
 819:                 omp_get_thread_num(), bs, n, n, ss, 0, ss, ss+bs-1, 0, ss-1 
 820:               );
 821:          }
 822: #endif
 823:          matrix_mult(bs, n, ss, A, ss, 0, size, B, 0, 0, size, C, ss, 0, size, 1);
 824:           }
 825: 
 826:           /* compute c22 <= (a21,a22) x (b12,b22)T */
 827:           #pragma omp task
 828:           {
 829: #if DEBUG_PAR 
 830:          #pragma omp critical(debug)
 831:          {
 832:               printf("task: (t%d) (%dx%d)*(%dx%d); depth %d; c22[%d..%d][%d..%d]\n",
 833:                 omp_get_thread_num(), bs, n, n, bs, 0, 0, bs-1, 0, bs-1 
 834:               );
 835:          }
 836: #endif
 837:          matrix_mult(bs, n, bs, A, ss, 0, size, B, 0, ss, size, C, ss, ss, size, 1);
 838:           }
 839:       }
 840: 
 841:       /* compute c11 <= c11 + tmp */
 842:       matrix_add(ss, ss, C, 0, 0, size, tmp, 0, 0, ss, C, 0, 0, size);
 843:        }
 844:    }
 845:     }
 846: }

 849: int 
 850: main(int argc, char *argv[])
 851: { 
 852:     int i, j, r, qdepth, numthreads, fulldepth;
 853: #ifndef NO_TIMERS
 854:     double start, end, total = 0.0;
 855:     double start_cal, end_cal, total_cal = 0.0;
 856: #endif
 857:     double rel_error, max_error;
 858:     double *A, *B, *C, *D;
 859: 
 860:     process_args(argc, argv, &size, &rep, &taskq, &strass, &block, &threadreq, &parallel);
 861: 
 862:     /* for allocation of thread stacks */
 863:     if (parallel) kmp_set_stacksize( max( (3*size*size + 128*size) * sizeof(double), 1024*1024 ) );
 864: 
 865:     /* Workaround for need to set vm-vpagemax in /etc/sysconfigtab to huge value for Tru64 v4.0 */
 866:     #pragma omp parallel
 867:     ;
 868: 
 869:     /* allocate array storage */
 870:     A = (double *) malloc (size * size * sizeof(double));
 871:     B = (double *) malloc (size * size * sizeof(double));
 872:     C = (double *) malloc (size * size * sizeof(double));
 873:     D = (double *) malloc (size * size * sizeof(double));
 874: 
 875:     /* set up input matrices with random values */
 876:     srand((unsigned int)time(NULL));
 877: 
 878:     #pragma omp parallel for private(j)
 879:     for (i = 0; i < size; i++) {
 880:    for (j = 0; j < size; j++) {
 881:        A[i*size + j] = ((float) rand()) / ((float) RAND_MAX);
 882:        B[i*size + j] = ((float) rand()) / ((float) RAND_MAX);
 883: #if DEBUG_INPUT
 884:        if (i == j) {
 885:           #pragma omp critical(debug)
 886:       printf("A[%d][%d] = %18.16f, B[%d][%d] = %18.16f\n",
 887:          i, j, A[i*size+j], i, j, B[i*size+j]);
 888:        }
 889: #endif
 890:    }
 891:     }
 892: 
 893:     /* calculate runtime parameters */
 894:     numthreads = omp_get_max_threads();
 895:     fulldepth = 0;
 896:     for(i=size; i>=strass; i>>=1) fulldepth++;
 897: 
 898:     if (parallel) {
 899:         if (threadreq == 0) threadreq = numthreads;
 900:    qdepth = min(taskq, fulldepth);
 901:     } else {
 902:         qdepth = 0; 
 903:     }
 904: 
 905:     printf("\n");
 906:     printf("Parameters:\n");
 907:     printf("  array size: %d x %d\n", size, size);
 908:     printf("  threads: %d\n", threadreq);
 909:     printf("  taskq depth: %d\n", qdepth);
 910:     printf("  repetitions: %d\n", rep);
 911:     printf("  strassen minimum: %d\n", strass);
 912:     printf("  matmul blocksize: %d\n", block);
 913:     printf("\n");
 914:     fflush(stdout);
 915: 
 916:     for (r = 1; r <= rep; r++) {
 917: 
 918:    /* printf("\nRepetition: %d\n", r); */
 919: 
 920:    if (!parallel) {
 921:        /* do regular matrix multiply A * B => D for result comparision */
 922: #ifndef NO_TIMERS
 923:        elapse_(&start_cal);
 924: #endif
 925:        matrix_mult(size, size, size, A, 0, 0, size, B, 0, 0, size, D, 0, 0, size, 0);
 926: 
 927: #ifndef NO_TIMERS
 928:        elapse_(&end_cal);
 929:        total_cal += end_cal - start_cal;
 930: #endif
 931:    } else {
 932: 
 933:        /* do strassen serial matrix multiply A * B => C */
 934:        parallel = FALSE;
 935: #ifndef NO_TIMERS
 936:        elapse_(&start_cal);
 937: #endif	
 938:        strassen_main(size, A, B, D, qdepth, strass, fulldepth);
 939: 
 940: #ifndef NO_TIMERS
 941:        elapse_(&end_cal);
 942:        total_cal += end_cal - start_cal;
 943: #endif	    
 944:        parallel = TRUE;
 945:    }
 946: 
 947:    /* do strassen parallel matrix multiply A * B => C */
 948: #ifndef NO_TIMERS
 949:         elapse_(&start);
 950: #endif
 951:    strassen_main(size, A, B, C, qdepth, strass, fulldepth);
 952: 
 953: #ifndef NO_TIMERS
 954:         elapse_(&end);
 955:    total += end - start;
 956: #endif
 957: 
 958:    /* calculate largest relative error */
 959:    max_error = 0.0;
 960:    for (i = 0; i < size; i++) {
 961:        for (j = 0; j < size; j++) {
 962:            rel_error = abs(C[i*size + j] - D[i*size + j]) /
